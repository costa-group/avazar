func izero(in:ff) -> out:ff {
  if (in == 0) {
    inv = 0
  } else {
    inv = felt.div 1 in
  }
  aux = felt.mul in inv
  out = felt.sub 1 aux
}

func main(%x:ff) -> %y:ff {
  call izero(%x) to %y
}
